\textbf{func} $modexp(a,b,n)$:
\par
\textbf{if} $b=0$ \textbf{then} return 1
\par
z = $modexp(a,\lfloor b/2 \rfloor,n)$
\par
\textbf{if} $b$ is odd \textbf{then} return $z^2$ (mod $n$)
\par
\textbf{else} return $z^2 \cdot a$ (mod $n$)
\hrule

\textbf{func} $extEuclid(a,b)$:
\par
\hspace{10pt} \textbf{if} $b=0$ \textbf{then} return $(1,0,a)$
\par
$x,y,z = extEuclid(b,a \mod b)$
\par
\hspace{10pt} return $(y,x - \lfloor a/b \rfloor y,z)$

\par
\hspace{10pt} T = null
\par
\hspace{10pt} H.$makequeue(S)$
\par
\hspace{10pt} \textbf{while} $|H|>1$ \textbf{do}
\par
\hspace{20pt} i $\gets$ H.$deletemin()$
\par
\hspace{20pt} j $\gets$ H.$deletemin()$
\par
\hspace{20pt} $f(k) \gets f(i)+f(j)$
\par
\hspace{20pt} T.$insert(k)$
\par
\hspace{20pt} T.$insert(i,j)$
\par
\hspace{20pt} $k.left \gets i$
\par
\hspace{20pt} $k.right \gets j$
\par
\hspace{10pt} \textbf{return} T
